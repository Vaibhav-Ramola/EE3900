	\problem Consider an LTI system with frequency response:\\
	
	$ H(e^{j\omega}) = e^{-j(\omega - \pi/4)}\left( \frac{1 + e^{-j2\omega} + 4e^{-j4\omega}}{1 + \frac{1}{2}e^{-j2\omega}} \right) $, \quad $-\pi < \omega < \pi$,\\
	
	Determine output $y[n]$ for all $n$ if the input for all $n$ is :
	\begin{align*}
	x[n] = \cos(\frac{\pi n}{2})
	\end{align*}
	\solution
	\begin{align*}
	H(e^{j\omega}) &= e^{j\pi/4} \cdot e^{-j\omega} \left( \frac{1 + e^{-j2\omega} + 4e^{-j4\omega}}{1 + \frac{1}{2}e^{-j2\omega}} \right) \\
	&= e^{j\pi/4}G(e^{j\omega})
	\end{align*}
	Let $y_{1}[n] = x[n] * g[n]$, then
	\begin{align*}
	x[n] &= \cos \left( \frac{n\pi}{2} \right) = \frac{e^{j\pi n/2} + e^{-j\pi n/2}}{2}\\
	y_{1}[n] &= \frac{G(e^{j\pi/2})e^{j\pi n/2} + G(e^{-j\pi/2}) e^{j\pi n/2} }{2}\\
	\end{align*}
	Evaluating the frequency response at $\omega = \pm \pi/2$ :
	\begin{align*}
	G(e^{j\pi/2}) &= e^{-j\pi/2} \frac{1 + e^{-j\pi} + 4e^{-j2\pi}}{1 + \frac{1}{2} e^{-j\pi}} = 8e^{-j\pi/2}\\
	G(e^{-j\pi/2}) &= 8e^{j\pi/2}
	\end{align*}
	Therefore,
	\begin{align*}
	y_{1}[n] &= (8 e^{j(\pi n/2 - \pi/2)} + 8e^{j(-\pi n/2 + \pi/2)})/2 = 8\cos \left(\frac{\pi}{2}n - \frac{\pi}{2}\right)
	\end{align*}
	and\\
	$ y[n] = e^{j \pi/4}y_{1}[n] = 8e^{j\pi/4}\cos(\frac{\pi}{2}n - \frac{\pi}{2}) $\\
